from pydantic import BaseModel, Field

from typing import Literal

from datetime import datetime

import uuid

 

class Order(BaseModel):

  order_id: str = Field(

    default_factory=lambda: str(uuid.uuid4()))

  customer_id: str

  product_id: str

  qty: int = Field(ge=1, le=100)

  total: float = Field(ge=0)

  status: Literal[

    'delivered','shipped',

    'processing','cancelled']

  payment: Literal[

    'pix','credit_card','boleto']

  created_at: datetime

 

  @validator('total')

  def total_must_be_reasonable(cls, v):

    if v > 50000:

      raise ValueError('Total acima do limite')

    return round(v, 2)

Field(ge=1, le=100)

Constraints numéricas: ge=maior-ou-igual, le=menor-ou-igual
